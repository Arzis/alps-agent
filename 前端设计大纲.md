# 企业级智能问答助手 前端设计方案

> **框架选择**: React 18 + TypeScript + Ant Design Pro
> **设计语言**: Ant Design 5.x
> **状态管理**: Zustand + React Query
> **可视化**: ECharts + Ant Design Charts
> **实时通信**: WebSocket (原生) + SSE (EventSource)

------

## 一、前端架构总览

### 1.1 技术栈全景

YAML



```
# === 核心框架 ===
框架:            React 18.3+ (TypeScript 5.5+)
构建工具:        Vite 5.x
路由:            React Router 6.x
UI组件库:        Ant Design 5.x + @ant-design/pro-components
CSS方案:         TailwindCSS 3.x + CSS Modules (混合使用)

# === 状态管理 ===
客户端状态:      Zustand 4.x (轻量, 替代Redux)
服务端状态:      TanStack React Query 5.x (接口缓存/同步)
表单管理:        React Hook Form + Zod (校验)

# === 数据可视化 ===
图表:            ECharts 5.x (via echarts-for-react)
指标卡片:        @ant-design/charts
Markdown渲染:    react-markdown + rehype-highlight
代码高亮:        Prism.js

# === 实时通信 ===
SSE:             原生 EventSource API (流式对话)
WebSocket:       原生 WebSocket API (审查通知)

# === 工程化 ===
代码规范:        ESLint + Prettier
Git Hooks:       Husky + lint-staged
测试:            Vitest + React Testing Library
E2E测试:         Playwright
国际化:          react-i18next
```

### 2.1 路由结构

React



```
// src/config/routes.tsx

import { lazy } from 'react';
import {
  MessageOutlined,
  FileTextOutlined,
  CheckCircleOutlined,
  BarChartOutlined,
  DashboardOutlined,
  EditOutlined,
  CommentOutlined,
  ShareAltOutlined,
  SettingOutlined,
} from '@ant-design/icons';

export interface RouteConfig {
  path: string;
  name: string;
  icon?: React.ReactNode;
  component: React.LazyExoticComponent<any>;
  layout?: 'main' | 'chat' | 'auth' | 'none';
  roles?: string[];        // 允许的角色
  hideInMenu?: boolean;
  children?: RouteConfig[];
}

const routes: RouteConfig[] = [
  // === 对话 (独立布局) ===
  {
    path: '/chat',
    name: '智能问答',
    icon: <MessageOutlined />,
    component: lazy(() => import('@/pages/Chat')),
    layout: 'chat',
  },

  // === 文档管理 ===
  {
    path: '/documents',
    name: '文档管理',
    icon: <FileTextOutlined />,
    component: lazy(() => import('@/pages/Documents')),
    layout: 'main',
  },

  // === 审查管理 ===
  {
    path: '/review',
    name: '审查管理',
    icon: <CheckCircleOutlined />,
    component: lazy(() => import('@/pages/Review')),
    layout: 'main',
    roles: ['admin', 'reviewer'],
    children: [
      {
        path: '/review/queue',
        name: '审查队列',
        component: lazy(() => import('@/pages/Review/ReviewQueue')),
      },
      {
        path: '/review/:id',
        name: '审查详情',
        component: lazy(() => import('@/pages/Review/ReviewDetail')),
        hideInMenu: true,
      },
      {
        path: '/review/stats',
        name: '审查统计',
        component: lazy(() => import('@/pages/Review/ReviewStats')),
      },
    ],
  },

  // === 评估中心 ===
  {
    path: '/evaluation',
    name: '评估中心',
    icon: <BarChartOutlined />,
    component: lazy(() => import('@/pages/Evaluation')),
    layout: 'main',
    roles: ['admin'],
  },

  // === 系统监控 ===
  {
    path: '/monitoring',
    name: '系统监控',
    icon: <DashboardOutlined />,
    component: lazy(() => import('@/pages/Monitoring')),
    layout: 'main',
    roles: ['admin'],
  },

  // === Prompt管理 ===
  {
    path: '/prompts',
    name: 'Prompt管理',
    icon: <EditOutlined />,
    component: lazy(() => import('@/pages/Prompts')),

    path: '/feedback',
    name: '反馈分析',
    icon: <CommentOutlined />,
```

```
// src/hooks/useChat.ts

import { useState, useCallback, useRef, useEffect } from 'react';
import { useChatStore } from '@/stores/chatStore';
import { chatService } from '@/services/chatService';
import { ChatMessage, StreamEvent } from '@/types/chat';

export function useChat(sessionId: string | null) {
  const {
    messages, setMessages, addMessage, updateLastMessage,
    sessionStatus, setSessionStatus,
  } = useChatStore();

  const [isStreaming, setIsStreaming] = useState(false);
  const abortControllerRef = useRef<AbortController | null>(null);

  // 加载历史消息
  useEffect(() => {
    if (sessionId) {
      loadHistory(sessionId);
    }
  }, [sessionId]);

  const loadHistory = async (sid: string) => {
    try {
      const history = await chatService.getHistory(sid);
      setMessages(history.messages);
      setSessionStatus(history.status || 'active');
    } catch (error) {
      // 新会话, 无历史
      setMessages([]);
    }
  };

  const sendMessage = useCallback(async (
    content: string,
    collection: string = 'default',
  ) => {
    if (!content.trim() || isStreaming) return;

    const newSessionId = sessionId || `sess_${Date.now()}`;

    // 添加用户消息
    const userMsg: ChatMessage = {
      id: Date.now(),
      role: 'user',
      content,
      timestamp: new Date().toISOString(),
    };
    addMessage(userMsg);

    // 添加空的助手消息 (流式填充)
    const assistantMsg: ChatMessage = {
      id: Date.now() + 1,
      role: 'assistant',
      content: '',
      timestamp: new Date().toISOString(),
      metadata: {},
    };
    addMessage(assistantMsg);

    setIsStreaming(true);
    abortControllerRef.current = new AbortController();

    try {
      // SSE 流式请求
      const eventSource = chatService.streamChat({
        message: content,
        session_id: newSessionId,
        collection,
        stream: true,
      });

      eventSource.onmessage = (event: MessageEvent) => {
        const data = JSON.parse(event.data);

        switch (data.event || event.type) {
          case 'token':
            // 逐token追加到最后一条助手消息
            updateLastMessage((prev) => ({
              ...prev,
              content: prev.content + data.data,
            }));
            break;

          case 'citation':
            updateLastMessage((prev) => ({
              ...prev,
              citations: JSON.parse(data.data),
            }));
            break;

          case 'status':
            const statusData = JSON.parse(data.data);
            if (statusData.status === 'waiting_review') {
              setSessionStatus('waiting_review');
            }
            break;

          case 'done':
            const doneData = JSON.parse(data.data);
            updateLastMessage((prev) => ({
              ...prev,
              confidence: doneData.confidence,
              model_used: doneData.model_used,
              fallback_used: doneData.fallback_used,
              metadata: {
                ...prev.metadata,
                cache_hit: doneData.cache_hit,
                latency_ms: doneData.latency_ms,
                tokens_used: doneData.tokens_used,
              },
            }));
            setIsStreaming(false);
            eventSource.close();
            break;

          case 'error':
            updateLastMessage((prev) => ({
              ...prev,
              content: `⚠️ 发生错误: ${data.data}`,
            }));
            setIsStreaming(false);
            eventSource.close();
            break;
        }
      };

      eventSource.onerror = () => {
        setIsStreaming(false);
        eventSource.close();
      };

    } catch (error) {
      console.error('Chat error:', error);
      updateLastMessage((prev) => ({
        ...prev,
        content: '⚠️ 请求失败, 请稍后重试。',
      }));
      setIsStreaming(false);
    }
  }, [sessionId, isStreaming, addMessage, updateLastMessage]);

  const stopStreaming = useCallback(() => {
    abortControllerRef.current?.abort();
    setIsStreaming(false);
  }, []);

  return {
    messages,
    isStreaming,
    sessionStatus,
    sendMessage,
    stopStreaming,
  };
}
```

```
          notification.success({
            message: '审查完成',
            description: '您的问题已得到审查确认',
          });
        }
      } catch (e) {
        console.error('WebSocket message parse error:', e);
      }
    };

    ws.onclose = () => {
      setConnected(false);
      // 自动重连
      const maxReconnects = config.maxReconnects ?? 10;
      if (reconnectCountRef.current < maxReconnects) {
        reconnectCountRef.current++;
        setTimeout(connect, config.reconnectInterval ?? 5000);
      }
    };

    ws.onerror = (error) => {
      console.error('WebSocket error:', error);
    };

    wsRef.current = ws;

    // 心跳
    const heartbeat = setInterval(() => {
      if (ws.readyState === WebSocket.OPEN) {
        ws.send('ping');
      }
    }, 30000);

      clearInterval(heartbeat);
      ws.close();
    };
  }, [token, user, config.url]);

    const cleanup = connect();
    return () => cleanup?.();
  }, [connect]);

  return { connected, ws: wsRef.current };
```

```
// src/hooks/useSSE.ts

import { useRef, useCallback, useState } from 'react';
import { useAuthStore } from '@/stores/authStore';

interface SSEConfig {
  onToken: (token: string) => void;
  onCitation?: (citations: any[]) => void;
  onDone?: (data: any) => void;
  onError?: (error: string) => void;
  onStatus?: (data: any) => void;
}

export function useSSE() {
  const [isStreaming, setIsStreaming] = useState(false);
  const abortRef = useRef<AbortController | null>(null);

  const startStream = useCallback(async (
    url: string,
    body: any,
    config: SSEConfig,
  ) => {
    setIsStreaming(true);
    abortRef.current = new AbortController();
    const token = useAuthStore.getState().token;

    try {
      const response = await fetch(url, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${token}`,
        },
        body: JSON.stringify(body),
        signal: abortRef.current.signal,
      });

      if (!response.ok) {
        throw new Error(`HTTP ${response.status}`);
      }

      const reader = response.body?.getReader();
      const decoder = new TextDecoder();

      if (!reader) throw new Error('No reader');

      let buffer = '';

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        buffer += decoder.decode(value, { stream: true });
        const lines = buffer.split('\n');
        buffer = lines.pop() || '';

        for (const line of lines) {
          if (line.startsWith('data: ')) {
            const data = line.slice(6);
            if (data === '[DONE]') {
              setIsStreaming(false);
              return;
            }

            try {
              const parsed = JSON.parse(data);

              switch (parsed.event) {
                case 'token':
                  config.onToken(parsed.data);
                  break;
                case 'citation':
                  config.onCitation?.(JSON.parse(parsed.data));
                  break;
                case 'done':
                  config.onDone?.(JSON.parse(parsed.data));
                  setIsStreaming(false);
                  break;
                case 'status':
                  config.onStatus?.(JSON.parse(parsed.data));
                  break;
                case 'error':
                  config.onError?.(parsed.data);
                  setIsStreaming(false);
                  break;
              }
            } catch (e) {
              // 可能是纯文本token
              config.onToken(data);
            }
          }
        }
      }
    } catch (error: any) {
      if (error.name !== 'AbortError') {
        config.onError?.(error.message);
      }
    } finally {
      setIsStreaming(false);
    }
  }, []);

  const stopStream = useCallback(() => {
    abortRef.current?.abort();
    setIsStreaming(false);
  }, []);

  return { startStream, stopStream, isStreaming };
}
```

### 9.2 Nginx 配置

nginx



```
# frontend/deploy/nginx.conf

server {
    listen 80;
    server_name _;
    root /usr/share/nginx/html;
    index index.html;

    # Gzip压缩
    gzip on;
    gzip_types text/plain text/css application/json application/javascript text/xml;
    gzip_min_length 1024;

    # 静态资源缓存
    location /assets/ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    # API代理
    location /api/ {
        proxy_pass http://qa-api:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_read_timeout 120s;
    }

    # WebSocket代理
    location /ws/ {
        proxy_pass http://qa-api:8000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_read_timeout 86400;
    }

    # SSE代理 (长连接)
    location /api/v1/chat/completions/stream {
        proxy_pass http://qa-api:8000;
        proxy_set_header Connection '';
        proxy_http_version 1.1;
        proxy_buffering off;
        proxy_cache off;
        proxy_read_timeout 120s;
        chunked_transfer_encoding off;
    }

    # SPA路由 fallback
    location / {
        try_files $uri $uri/ /index.html;
    }
}
```
